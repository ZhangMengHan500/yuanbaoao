# AI出卷功能实现计划

> **面向 AI 代理的工作者：** 必需子技能：使用 superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans 逐任务实现此计划。步骤使用复选框（`- [ ]`）语法来跟踪进度。

**目标：** 实现仿腾讯元宝的 AI 试卷生成功能，包含相似卷、错题巩固卷、自定义试卷三种模式，支持上传/拍摄试卷图片和语音输入。

**架构：** 前端 React Native 新增 AiExamScreen，通过 ChatActionPanel 的「AI出卷」入口进入。后端 NestJS 新增 exam 模块，负责文件上传、调用 OCR 服务解析试卷、调用 Python AI 服务生成试卷。Python 服务新增 FastAPI 端点处理试卷解析和生成。

**技术栈：** React Native (Expo), TypeScript, NestJS, Python FastAPI, PaddleOCR, 大模型 API

---

## 整体架构

```
用户点击「AI出卷」→ AiExamScreen
  ↓
选择模式：相似卷 / 错题巩固卷 / 自定义试卷
  ↓
上传/拍摄试卷图片 → POST /exam/upload (NestJS)
  ↓
NestJS 调用 OCR 服务 (localhost:5000) → 提取题目文本
  ↓
NestJS 调用 Python AI 服务 (localhost:8000) → 生成试卷
  ↓
流式返回生成结果 → 前端渲染试卷
```

### exam.controller.ts

```typescript
import {
  Controller, Post, Body, UploadedFile, UseInterceptors,
  UseGuards, Res, BadRequestException,
} from '@nestjs/common';
import { FileInterceptor } from '@nestjs/platform-express';
import { Response } from 'express';
import { ExamService } from './exam.service';
import { JwtAuthGuard } from '../common/guards/jwt-auth.guard';
import { CurrentUser } from '../common/decorators/current-user.decorator';
import { IsString, IsOptional, IsNumber, IsArray } from 'class-validator';

class ParseExamDto {
  @IsString()
  imageUrl: string;
}

class GenerateSimilarDto {
  @IsString()
  imageUrl: string;
  @IsOptional()
  @IsNumber()
  count?: number;
}

class GenerateReviewDto {
  @IsString()
  imageUrl: string;
  @IsOptional()
  @IsArray()
  weakPoints?: string[];
}

class GenerateCustomDto {
  @IsOptional()
  @IsString()
  subject?: string;
  @IsOptional()
  @IsString()
  grade?: string;
  @IsOptional()
  @IsString()
  questionTypes?: string;
  @IsOptional()
  @IsNumber()
  questionCount?: number;
  @IsOptional()
  @IsString()
  difficulty?: string;
  @IsOptional()
  @IsString()
  knowledgePoints?: string;
  @IsOptional()
  @IsString()
  description?: string;
}

@Controller('exam')
@UseGuards(JwtAuthGuard)
export class ExamController {
  constructor(private examService: ExamService) {}

  // 上传试卷图片
  @Post('upload')
  @UseInterceptors(FileInterceptor('file'))
  async uploadExam(
    @CurrentUser('id') userId: string,
    @UploadedFile() file: Express.Multer.File,
  ) {
    if (!file) throw new BadRequestException('请上传试卷图片');
    return {
      url: `/uploads/exam/${file.filename}`,
      filename: file.filename,
      size: file.size,
    };
  }

  // 解析试卷图片
  @Post('parse')
  async parseExam(
    @CurrentUser('id') userId: string,
    @Body() dto: ParseExamDto,
    @Res() res: Response,
  ) {
    res.setHeader('Content-Type', 'text/event-stream');
    res.setHeader('Cache-Control', 'no-cache');
    res.setHeader('Connection', 'keep-alive');
    res.flushHeaders();

    try {
      const content = await this.examService.parseExamImage(dto.imageUrl);
      res.write(`data: ${JSON.stringify({ type: 'done', content })}\n\n`);
      res.end();
    } catch (error: any) {
      res.write(`data: ${JSON.stringify({ type: 'error', message: error.message })}\n\n`);
      res.end();
    }
  }

  // 生成相似卷（SSE 流式）
  @Post('generate/similar')
  async generateSimilar(
    @CurrentUser('id') userId: string,
    @Body() dto: GenerateSimilarDto,
    @Res() res: Response,
  ) {
    res.setHeader('Content-Type', 'text/event-stream');
    res.setHeader('Cache-Control', 'no-cache');
    res.setHeader('Connection', 'keep-alive');
    res.flushHeaders();

    try {
      const content = await this.examService.parseExamImage(dto.imageUrl);
      for await (const token of this.examService.generateSimilarExam(content, dto.count)) {
        res.write(`data: ${JSON.stringify({ token, done: false })}\n\n`);
      }
      res.write(`data: ${JSON.stringify({ done: true })}\n\n`);
      res.end();
    } catch (error: any) {
      res.write(`data: ${JSON.stringify({ error: error.message })}\n\n`);
      res.end();
    }
  }

  // 生成错题巩固卷（SSE 流式）
  @Post('generate/review')
  async generateReview(
    @CurrentUser('id') userId: string,
    @Body() dto: GenerateReviewDto,
    @Res() res: Response,
  ) {
    res.setHeader('Content-Type', 'text/event-stream');
    res.setHeader('Cache-Control', 'no-cache');
    res.setHeader('Connection', 'keep-alive');
    res.flushHeaders();

    try {
      const content = await this.examService.parseExamImage(dto.imageUrl);
      for await (const token of this.examService.generateReviewExam(content, dto.weakPoints || [])) {
        res.write(`data: ${JSON.stringify({ token, done: false })}\n\n`);
      }
      res.write(`data: ${JSON.stringify({ done: true })}\n\n`);
      res.end();
    } catch (error: any) {
      res.write(`data: ${JSON.stringify({ error: error.message })}\n\n`);
      res.end();
    }
  }

  // 生成自定义试卷（SSE 流式）
  @Post('generate/custom')
  async generateCustom(
    @CurrentUser('id') userId: string,
    @Body() dto: GenerateCustomDto,
    @Res() res: Response,
  ) {
    res.setHeader('Content-Type', 'text/event-stream');
    res.setHeader('Cache-Control', 'no-cache');
    res.setHeader('Connection', 'keep-alive');
    res.flushHeaders();

    try {
      for await (const token of this.examService.generateCustomExam(dto)) {
        res.write(`data: ${JSON.stringify({ token, done: false })}\n\n`);
      }
      res.write(`data: ${JSON.stringify({ done: true })}\n\n`);
      res.end();
    } catch (error: any) {
      res.write(`data: ${JSON.stringify({ error: error.message })}\n\n`);
      res.end();
    }
  }
}
```

```python
"""
AI出卷服务 - 独立 FastAPI 服务（可选）
如果不需要独立部署，可使用 NestJS 中的 ExamService 直接调用大模型
"""
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from typing import Optional, List
import httpx
import os

app = FastAPI(title="AI出卷服务")

# ========== 配置 ==========
# 替换为你的真实大模型 API
LLM_API_URL = os.getenv("LLM_API_URL", "https://api.deepseek.com/v1/chat/completions")
LLM_API_KEY = os.getenv("LLM_API_KEY", "")
LLM_MODEL = os.getenv("LLM_MODEL", "deepseek-chat")

OCR_SERVICE_URL = os.getenv("OCR_SERVICE_URL", "http://localhost:5000")


class ParseRequest(BaseModel):
    image_base64: str


class GenerateRequest(BaseModel):
    mode: str  # similar / review / custom
    content: Optional[dict] = None
    weak_points: Optional[List[str]] = None
    params: Optional[dict] = None


@app.get("/health")
async def health():
    return {"status": "ok", "service": "exam-service"}


@app.post("/parse")
async def parse_exam(req: ParseRequest):
    """调用 OCR 服务解析试卷图片"""
    try:
        async with httpx.AsyncClient(timeout=30) as client:
            resp = await client.post(
                f"{OCR_SERVICE_URL}/ocr",
                json={"image_base64": req.image_base64},
            )
            return resp.json()
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"OCR服务调用失败: {str(e)}")


@app.post("/generate")
async def generate_exam(req: GenerateRequest):
    """调用大模型生成试卷"""
    if req.mode == "similar":
        prompt = build_similar_prompt(req.content)
    elif req.mode == "review":
        prompt = build_review_prompt(req.content, req.weak_points)
    elif req.mode == "custom":
        prompt = build_custom_prompt(req.params)
    else:
        raise HTTPException(status_code=400, detail="未知的生成模式")

    try:
        async with httpx.AsyncClient(timeout=60) as client:
            resp = await client.post(
                LLM_API_URL,
                headers={"Authorization": f"Bearer {LLM_API_KEY}"},
                json={
                    "model": LLM_MODEL,
                    "messages": [{"role": "user", "content": prompt}],
                    "temperature": 0.7,
                },
            )
            data = resp.json()
            return {"result": data["choices"][0]["message"]["content"]}
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"生成失败: {str(e)}")


def build_similar_prompt(content: dict) -> str:
    questions = content.get("questions", [])
    subject = content.get("subject", "数学")
    grade = content.get("grade", "")
    q_text = "\n".join([f"{q['index']}. [{q['type']}] {q['content']}" for q in questions])
    return f"基于以下{subject}试卷，生成10道相似题目：\n\n{q_text}\n\n要求：保持相同知识点和难度，题目内容不同。"


def build_review_prompt(content: dict, weak_points: list) -> str:
    points = "、".join(weak_points) if weak_points else "综合巩固"
    questions = content.get("questions", [])
    q_text = "\n".join([f"{q['index']}. [{q['type']}] {q['content']}" for q in questions])
    return f"针对薄弱知识点【{points}】生成巩固练习卷：\n\n{q_text}\n\n要求：每个知识点至少2道题，难度递进。"


def build_custom_prompt(params: dict) -> str:
    subject = params.get("subject", "数学")
    grade = params.get("grade", "")
    count = params.get("questionCount", 10)
    desc = params.get("description", "")
    return f"请生成一份{grade}{subject}试卷，共{count}道题。\n\n{desc}"


if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8001)
```
